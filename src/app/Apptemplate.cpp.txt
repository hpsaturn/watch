MainMenuApplication::~MainMenuApplication() {
    
}
MainMenuApplication::MainMenuApplication() {
    
}
bool MainMenuApplication::Tick() {
    if (millis() > nextRedraw ) {
        nextRedraw=millis()+(1000/2);
        return true;
    }
    return false;
}
